Con eventos

const search = new Search({
    element: '#app',
    data: items
}).init();

search
    .on('init', () => console.log('Componente listo'))
    .on('search', (data) => {
        console.log(`Buscando: ${data.searchTerm}`);
        console.log(`Resultados: ${data.results.length}`);
    })
    .on('itemSelected', ({ item, close }) => {
        alert(`Seleccionaste: ${item.nombre}`);
        close();
    })
    .on('pageChange', (data) => {
        console.log(`Página ${data.page}/${data.totalPages}`);
    });

Con caché y sort

const search = new Search({
    element: '#app',
    data: productos,
    cacheEnabled: true,
    cacheMaxSize: 100,
    cacheTtlSeconds: 600,
    sortBy: 'precio',
    sortOrder: 'asc'
}).init();

// Ordenar dinámicamente
document.getElementById('btn-ordenar').addEventListener('click', () => {
    search.sort('nombre', 'asc');
});

// Limpiar orden
document.getElementById('btn-reset').addEventListener('click', () => {
    search.clearSort();
    search.draw('');
});

// Reseteear estado completo
document.getElementById('btn-clear').addEventListener('click', () => {
    search.clear();
});

Integración con frameworks

React

import { useEffect, useRef } from 'react';
import { Search } from 'buscador-dinamico';
import 'buscador-dinamico/css/buscador-dinamico.css';

function SearchComponent({ data }) {
    const containerRef = useRef(null);
    const searchRef = useRef(null);

    useEffect(() => {
        searchRef.current = new Search({
            element: '#react-search',
            data: data,
            theme: 'clean-white'
        }).init();

        return () => {
            searchRef.current?.destroy();
        };
    }, [data]);

    return <div id="react-search" ref={containerRef}></div>;
}

Vue

<template>
    <div id="vue-search"></div>
</template>

<script setup>
import { onMounted, onUnmounted } from 'vue';
import { Search } from 'buscador-dinamico';
import 'buscador-dinamico/css/buscador-dinamico.css';

const props = defineProps({ data: Array });
let search = null;

onMounted(() => {
    search = new Search({
        element: '#vue-search',
        data: props.data,
        theme: 'clean-white'
    }).init();
});

onUnmounted(() => {
    search?.destroy();
});
</script>
